编译安装Aerospike

# 编译安装Aerospike

# 前言

Aerospike专门为SSD/Flash设备优化，为集群计算与跨数据中心设计。目标是对大数据、集群计算的应用，在全部数据的范围内提供内存级的高速度，以及企业级的可扩展性和可靠性。

# 编译

  • 安装依赖包
yum install -y openssl openssl-devel openssl-static lua lua-devel lua-static libtool
1
  • git clone https://github.com/aerospike/aerospike-server.git
  • git submodule update --init
  • make all -j8
  • 创建rpm包 - make rpm
  • 安装 - rpm -ivh aerospike-server-community-4.7.0.3-1.el7.x86_64.rpm
  • 下载tools - wget -O aerospike-tools.tgz 'https://www.aerospike.com/download/tools/latest/artifact/el7'
  • 安装tools - rpm -ivh aerospike-tools-3.8.2-1.el7.x86_64.rpm

# 初始化

  • 编辑/etc/aerospike/aerospike.conf
# Aerospike database configuration file.

service {
        user root
        group root
        paxos-single-replica-limit 1 # Number of nodes where the replica count is automatically reduced to 1.
        pidfile /var/run/aerospike/asd.pid
        service-threads 4
        transaction-queues 4
        transaction-threads-per-queue 4
        proto-fd-max 15000
}

logging {
        # Log file must be an absolute path.
        file /var/log/aerospike/aerospike.log {
                context any info
        }
}

network {
        service {
                address any
                port 3000
        }

        heartbeat {
                mode mesh
                port 3002 # Heartbeat port for this node.

                # List one or more other nodes, one ip-address & port per line:
                mesh-seed-address-port 12.0.0.1 3002
                mesh-seed-address-port 12.0.0.2 3002
                mesh-seed-address-port 12.0.0.3 3002

                # To use unicast-mesh heartbeats, remove the 3 lines above, and see
                # aerospike_mesh.conf for alternative.

                interval 150
                timeout 10
        }

        fabric {
                port 3001
        }

        info {
                port 3003
        }
}

namespace testns {
        replication-factor 2
        memory-size 60G # 90% physical mem
        high-water-memory-pct 75
#       storage-engine memory
#        ldt-enabled   true
        # To use file storage backing, comment out the line above and use the
        # following lines instead.
        storage-engine device {
#               device  /dev/sdb1 # for ssd
                file /opt/aerospike/data/testns.dat
                filesize 200G # default is 16GB
#               write-block-size 128K
#               data-in-memory true # Store data in memory in addition to file.
        }
}
