通过源码编译安装PostgreSQL 9.6

# 通过源码编译安装PostgreSQL 9.6

# 操作系统配置

  1. /etc/sysctl.conf
vm.overcommit_memory = 2
vm.overcommit_ratio = 95
1
2
  1. /etc/security/limits.conf
@postgres      hard    nproc           655350
@postgres        soft    nproc           655350
*               soft    memlock         unlimited
*               hard    memlock         unlimited
* soft core unlimited
1
2
3
4
5
  1. /etc/rc.local
echo deadline >/sys/block/sda/queue/scheduler
echo never > /sys/kernel/mm/transparent_hugepage/enabled

2
  1. 添加用户
groupadd postgres 2>/dev/null
useradd -g postgres postgres 2>/dev/null
mkdir /usr/local/pgsql1921 /pgsql/pgsql1921 -p  2>/dev/null
chown postgres:postgres /usr/local/pgsql1921 /pgsql/pgsql1921 -R
chmod 700 /pgsql/pgsql1921

  1. 修改postgres用户环境变量
export PGPORT=1921

export LANG=en_US.utf8        
export PGHOME=/usr/local/pgsql1921
export LD_LIBRARY_PATH=$PGHOME/lib:/lib64:/usr/lib64:/usr/local/lib64:/lib:/usr/lib:/usr/local/lib:$LD_LIBRARY_PATH         
export PATH=$PGHOME/bin:$PATH:.        
export MANPATH=$PGHOME/share/man:$MANPATH        
export PGHOST=$PGDATA        
export PGUSER=postgres        
export PGDATABASE=postgres

  1. 安装依赖包

# 编译

* tar -xf postgresql-9.6.2.tar.gz
* cd postgresql-9.6.2
./configure --prefix=/usr/local/pgsql1921 \
--with-pgport=1921 \
--with-perl \
--with-tcl \
--with-python \
--with-openssl \
--with-pam \
--without-ldap \
--with-libxml \
--with-libxslt \
--enable-thread-safety \
--with-wal-segsize=64 \
--with-segsize=1
* gmake world -j8
* gmake install-world

# 设置访问权限

修改pg_hba.conf

# 流复制
host replication repl 0.0.0.0/0 md5
# 限制postgres用户远程登录
host all postgres 0.0.0.0/0 reject
# 其他用户需要密码验证
host all all 0.0.0.0/0 md5

# 配置系统服务

编辑/lib/systemd/system/pgsql5432.service

[Unit]
Description=PostgreSQL database server
After=network.target

[Service]
Type=forking

User=postgres
Group=postgres

# Port number for server to listen on
Environment=PGPORT=5432

# Location of database directory
Environment=PGDATA=/pgsql/pgsql5432

# Where to send early-startup messages from the server (before the logging
# options of postgresql.conf take effect)
# This is normally controlled by the global default set by systemd
# StandardOutput=syslog

# Disable OOM kill on the postmaster
OOMScoreAdjust=-1000

ExecStart=/usr/local/pgsql5432/bin/pg_ctl start -D ${PGDATA} -s -o "-p ${PGPORT}" -w -t 300
ExecStop=/usr/local/pgsql5432/bin/pg_ctl stop -D ${PGDATA} -s -m fast
ExecReload=/usr/local/pgsql5432/bin/pg_ctl reload -D ${PGDATA} -s

# Give a reasonable amount of time for the server to start up/shut down
TimeoutSec=300

[Install]
WantedBy=multi-user.target

# 创建用户

正常是新建一个组，然后给组授权，再新建用户加入组即可。

create role rwgroup nologin;
create user rwuser login encrypted password 'xxxxxx' IN ROLE rwgroup;
create role repl replication login encrypted password 'xxxxxx';
CREATE EXTENSION pg_stat_statements;"

# 授权

GRANT USAGE ON SCHEMA schema_name TO rwgroup;
GRANT SELECT ON ALL TABLES IN SCHEMA schema_name TO rwgroup;
ALTER DEFAULT PRIVILEGES IN SCHEMA myschema GRANT SELECT ON TABLES TO rwgroup;
