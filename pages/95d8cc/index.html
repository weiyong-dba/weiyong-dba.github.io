# 检查服务状态

#!/usr/bin/env python
# -*- coding: utf-8 -*-
import json
import argparse
import subprocess

metrics = []
if __name__ == "__main__":
    parser = argparse.ArgumentParser(description="检查scylla节点状态")
    down_node_count = subprocess.check_output("nodetool status | grep 'rack1' | grep -v '^UN' | wc -l",shell=True)
    count = (down_node_count.decode().strip().split('\n'))[0]
    metrics.append({"metric":"scylla.deadnode.count","data_type":"gauge","value":count,"tags":{"port":9160}})
    print json.dumps(metrics, indent=4)

# 恢复失败的节点

注意

如果是种子节点需要从/etc/scylla/scylla.yaml删除该节点的信息。

1.删除数据目录

rm -rf /var/lib/scylla/data/*
1

2.修改配置文件

auto_bootstrap: true
replace_address_first_boot: 172.xx.xx.xx
1
2

3.启动服务

service scylla-server start
1

4.修复 在修复节点执行repair

nodetool repair
1
