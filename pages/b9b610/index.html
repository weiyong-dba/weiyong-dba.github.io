通过源码编译Redis并制作RPM安装包

# 通过源码编译Redis并制作RPM安装包

# 前言

该rpm按以下约定完成安装：

  • 软件安装至/usr/local/redis6379目录
  • 服务属组为hadoop
  • 通过syctemctl实现服务自启动，服务名redis6379.service
  • 端口6379
  • 配置文件中在/usr/local/redis6379/redis.conf

# gcc依赖

redis6版本需要gcc5.3以上，redis4的话gcc4.8.5即可。

查看gcc版本是否在5.3以上，centos7.6默认安装4.8.5

gcc -v
1

升级gcc到5.3及以上,如下：

yum -y install centos-release-scl
yum -y install devtoolset-9-gcc devtoolset-9-gcc-c++ devtoolset-9-binutils
scl enable devtoolset-9 bash
1
2
3

需要注意的是scl命令启用只是临时的，退出shell或重启就会恢复原系统gcc版本。 如果要长期使用gcc 9.3的话：

echo "source /opt/rh/devtoolset-9/enable" >>/etc/profile
1

这样退出shell重新打开就是新版的gcc了。

# 安装工具包

yum install rpmbuild rpmdevtools -y
1

# 初始化目录

rpmdev-setuptree rpmbuild
1

# 准备源码包

把原码包redis-4.0.12.tar.gz上传到SOURCES目录下，其他需要用到的包也放在这个目录下。

# 编写spec文件

在SPECS目录下添加redis.spec文件，文件内容如下，可以根据需要修改。

Name:       redis-test
Version:    4.0.12
Release:    1.0.0
License:    GPL
URL:        http://www.redis.io
Group:      applications/database
Source:     redis-%{version}.tar.gz
BuildRoot:  %(mktemp -ud %{_tmppath}/%{name}-%{version}-%{release}-XXXXXX)
BuildRequires:  make,gcc
Packager:   weiyong@tendcloud.com
Autoreq:    no
Summary: Redis-Test-%{version} RPM

%description
Redis %{version} Server.

%define REDIS_PORT 6379
%define BASE_DIR /usr/local/redis%{REDIS_PORT}
%define REDIS_USER hadoop
%define REDIS_GROUP hadoop

%prep
%setup -q -n redis-%{version}

%build

%install
rm -fr %{buildroot}
make %{?_smp_mflags} -j `cat /proc/cpuinfo | grep processor| wc -l` PREFIX=%{buildroot}%{BASE_DIR} install
cp src/redis-trib.rb %{buildroot}%{BASE_DIR}/bin/

%pre
check_port() {
        netstat -nltup | grep -v "127.0.0.1" | grep $1 &>/dev/null
}
if [ -d %{BASE_DIR} ];then
        echo "Error: The base directory %{BASE_DIR} already exists, Aborting." ;
        exit 1
elif check_port %{REDIS_PORT}
then
        echo "Error: The port %{REDIS_PORT} already in used, Aborting." ;
        exit 1
fi

mkdir -p %{BASE_DIR}
#cp -r %{buildroot}%{BASE_DIR}/* %{BASE_DIR}/

cat > %{BASE_DIR}/redis.conf <<EOF
daemonize yes

port %{REDIS_PORT}

pidfile %{BASE_DIR}/redis.pid
# bind 127.0.0.1
# unixsocket /tmp/redis.sock
# unixsocketperm 755

timeout 0
loglevel verbose
logfile %{BASE_DIR}/stdout
databases 16

#save 86400 1
#save 86400 10
#save 86400 10000

rdbcompression yes
dbfilename dump.rdb
dir %{BASE_DIR}
slave-serve-stale-data yes
slave-priority 100
appendonly no

# appendfsync always
appendfsync everysec
no-appendfsync-on-rewrite no
auto-aof-rewrite-percentage 100
auto-aof-rewrite-min-size 64mb

slowlog-log-slower-than 10000
slowlog-max-len 1024

stop-writes-on-bgsave-error no

#vm-enabled no
#vm-swap-file /tmp/redis.swap
#vm-max-memory 0
#vm-page-size 32
#vm-pages 134217728
#vm-max-threads 4

#hash-max-zipmap-entries 512
#hash-max-zipmap-value 64

list-max-ziplist-entries 512
list-max-ziplist-value 64

set-max-intset-entries 512

zset-max-ziplist-entries 128
zset-max-ziplist-value 64

activerehashing yes

client-output-buffer-limit normal 0 0 0
client-output-buffer-limit slave 0 0 0
client-output-buffer-limit pubsub 33554432 8388608 60
repl-timeout 900

protected-mode no
lazyfree-lazy-server-del yes
lazyfree-lazy-eviction yes
lazyfree-lazy-expire yes
#cluster
#cluster-require-full-coverage no
#cluster-enabled yes
#cluster-config-file nodes.conf
#cluster-node-timeout 5000
#disable command
#rename-command KEYS ""
#rename-command FLUSHDB ""
#rename-command FLUSHALL ""
EOF

cat > %{BASE_DIR}/redis.server <<EOF
#!/bin/sh

REDISPORT="%{REDIS_PORT}"
EXEC=%{BASE_DIR}/bin/redis-server
CLIEXEC=%{BASE_DIR}/bin/redis-cli
PIDFILE=%{BASE_DIR}/redis.pid
CONF="%{BASE_DIR}/redis.conf"


case "\$1" in
    start)
        if [ -f \$\$PIDFILE ]
        then
                echo "\$PIDFILE exists, process is already running or crashed"
        else
                echo "Starting Redis server..."
                \$EXEC \$CONF
        fi
        ;;
    stop)
        if [ ! -f \$PIDFILE ]
        then
                echo "\$PIDFILE does not exist, process is not running"
        else
                PID=\$(cat \$PIDFILE)
                echo "Stopping ..."
                \$CLIEXEC -p \$REDISPORT shutdown
                while [ -x /proc/\${PID} ]
                do
                    echo "Waiting for Redis to shutdown ..."
                    sleep 1
                done
                echo "Redis stopped"
        fi
        ;;
    *)
        echo "Please use start or stop as first argument"
        ;;
esac
EOF

chown -R %{REDIS_USER}:%{REDIS_GROUP} %{BASE_DIR}
chmod 755 %{BASE_DIR}/redis.server

%post

printf "[Unit]\nDescription=redis%{REDIS_PORT}\n[Service]\nType=forking\nUser=%{REDIS_USER}\nGroup=%{REDIS_GROUP}\nExecStart=%{BASE_DIR}/redis.server start\nExecStop=%{BASE_DIR}/redis.server stop\nPrivateTmp=true\nTimeoutSec=60\n[Install]\nWantedBy=multi-user.target\n">/lib/systemd/system/redis%{REDIS_PORT}.service
systemctl enable redis%{REDIS_PORT}.service >/dev/null 2>&1
systemctl daemon-reload
systemctl start redis%{REDIS_PORT}.service >/dev/null 2>&1
sed -i 's/\/usr\/local\/redis6379\/bin://g' ~hadoop/.bash_profile
sed -i 's/\$PATH/\/usr\/local\/redis6379\/bin:\$PATH/g' ~hadoop/.bash_profile
echo "######################  redis 4.0.12  install successed ######################"
echo " VERSION = %{version}"
echo " BASEDIR = /usr/local/redis%{REDIS_PORT}"
echo " REDIS_CNF = /usr/local/redis%{REDIS_PORT}/redis.conf"
echo " USER = hadoop"
echo " PORT = %{REDIS_PORT}"
echo " Use su - hadoop;/usr/local/redis%{REDIS_PORT}/redis.srver start|stop to start or stop redis server"
echo " Run 'redis-cli -p %{REDIS_PORT} -h localhost' to connect!"

%preun
%{BASE_DIR}/redis.server stop >/dev/null 2>&1

%postun
rm -rf %{BASE_DIR}
rm -f /lib/systemd/system/redis%{REDIS_PORT}.service
sed -i 's/\/usr\/local\/redis6379\/bin://g' ~hadoop/.bash_profile

%clean
rm -rf %{buildroot}

%files
%defattr(-, %{REDIS_USER}, %{REDIS_GROUP})
%attr(755, %{REDIS_USER}, %{REDIS_GROUP}) %{BASE_DIR}/*

%changelog

# 开始制作

rpmbuild -bb rpmbuild/SPECS/redis.spec
1

完成后在rpmbuild/RPM目录下就会两个rpm包

-rw-r--r-- 1 root root 723032 Sep 3 17:14 redis-4.0.12-1.0.0.x86_64.rpm

-rw-r--r-- 1 root root 4371268 Sep 3 17:14 redis-debuginfo-4.0.12-1.0.0.x86_64.rpm

# 安装Redis

执行rpm安装，安装完成后会直接启动Redis：

[root@rhce ~]# rpm -ivh rpmbuild/RPMS/x86_64/redis-4.0.12-1.el6.x86_64.rpm
Preparing...                ############################# [100%]
   1:redis                  ############################# [100%]
Starting Redis server...
1
2
3
4

# 卸载Redis

不需要关闭服务直接卸载：

[root@rhce ~]# rpm -e --nodeps redis-4.0.12-1.el6.x86_64
Shutting down Redis..[  OK  ]
1
2

注意:

在卸载服务时会把redis目录删除，如果需要请先备份数据；

# 编译命令说明

rpmbuild

  • -ba 既生成src.rpm又生成二进制rpm
  • -bs 只生成src的rpm
  • -bb 只生二进制的rpm
  • -bp 执行到pre
  • -bc 执行到 build段
  • -bi 执行install段
  • -bl 检测有文件没包含
