数据迁移

# MySQL数据迁移至TiDB

# 导出数据

nohup ./dumpling \
  -u root \
  -p "abc" \
  -P 3307 \
  -B test \
  -T "test.charge" \
  -h 1.1.1.1 \
  --filetype sql \
  --escape-backslash false \
  -r 1000000 \
  -o /backup/test \
  --skip-tz-utc \
  --csv-separator ',' \
  --csv-delimiter '"' \
  -F 1GB &

导出效率

32亿数据，导出后原始文件大小800G，原始表大小1.5TB，导出耗时3小时，导入耗时3小时。

# 导入数据

注意：如果字段包含特殊字符时，导入出现各种报错 导入时先删除多余索引

  • 编辑tidb-lightning.toml
[mydumper]
data-source-dir = "/backup/test"
strict-format = true

[tidb]
host = "10.16.18.205"
port = 4000
user = "root"
password = ""

[tikv-importer]
sorted-kv-dir = "/backup/sorted"

[checkpoint]
# 是否启用断点续传。
# 导入数据时，TiDB Lightning 会记录当前表导入的进度。
# 所以即使 Lightning 或其他组件异常退出，在重启时也可以避免重复再导入已完成的数据。
enable = false

[mydumper.csv]
separator = ","
delimiter = '"'
# 是否解析字段内的反斜线转义符。
backslash-escape = true
# 是否移除以分隔符结束的行。
trim-last-separator = false
