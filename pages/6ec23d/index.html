from sqlalchemy import PrimaryKeyConstraint, UniqueConstraint, Index
from sqlalchemy.dialects.mysql import BIGINT,LONGTEXT
from app.extensions import db, ma

tablename = 'my_table'
id = db.Column(db.Integer, primary_key=True, autoincrement=True)
product_id = db.Column(db.Integer, index=True, comment="产品线ID")
engine_id = db.Column(db.CHAR(32), index=True)
table_schema = db.Column(db.String(64))
table_name = db.Column(db.VARCHAR(length=255,collation='utf8_bin'))
size = db.Column(BIGINT(unsigned=True))
mtime = db.Column(db.DateTime, comment="最后修改时间")
create_at = db.Column(db.DateTime, index=True, server_default=db.text("CURRENT_TIMESTAMP"), comment="创建时间")
table_args = (UniqueConstraint('engine_id', 'table_schema', 'table_name', name='uk_table_file_size'),
PrimaryKeyConstraint('instance_id', 'checksum', 'create_at', name='pk_global_query_review'),)
