  • 方式1

dict1_gt1 = {key: value for key, value in dict1.items()
                    if value > 1}
print(dict1_gt1)
1

3
4

dict1_gt2 = {x:dict1[x] for x in filter(lambda x:dict1[x] > 1,dict1)}

2
