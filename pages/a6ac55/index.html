    通过源码编译MySQL并制作RPM安装包

    # 安装工具包

    yum install rpmbuild rpmdevtools -y

    # 初始化目录

    rpmdev-setuptree rpmbuild

    # 准备源码包

    把原码包mysql-5.7.27.tar.gz上传到SOURCES目录下，其他需要用到的包也放在这个目录下。

    # 编写spec文件

    在SPECS目录下添加mysql.spec文件，文件内容如下，可以根据需要修改。

    Name:       mysql
Version:    5.7.27
Release:    1.0.0
License:    GPL
URL:        http://www.mysql.com
Group:      applications/database
Source0:     mysql-%{version}.tar.gz
Source1:     common_schema.sql
BuildRoot:  %(mktemp -ud %{_tmppath}/%{name}-%{version}-%{release}.el7.centos-XXXXXX)
BuildRequires:  cmake
Requires: libaio
Packager:   dba
Autoreq:    no
Summary: MySQL-%{version} RPM

%description
MySQL %{version} Server.

%define MYSQL_PORT 3306
%define DATA_DIR /mysql/mysql%{MYSQL_PORT}
%define BASE_DIR /usr/local/mysql%{MYSQL_PORT}
%define MYSQL_USER mysql
%define MYSQL_GROUP mysql

%prep
%setup -q mysql-%{version}

%build

cmake -DCMAKE_INSTALL_PREFIX=%{BASE_DIR} \
-DWITH_BOOST=boost/boost_1_59_0 \
-DMYSQL_DATADIR=%{DATA_DIR} \
-DSYSCONFDIR=%{BASE_DIR}/etc/ \
-DWITH_INNOBASE_STORAGE_ENGINE=1 \
-DWITH_ARCHIVE_STORAGE_ENGINE=1 \
-DWITH_FEDERATED_STORAGE_ENGINE=1 \
-DWITH_PARTITION_STORAGE_ENGINE=1 \
-DWITH_BLACKHOLE_STORAGE_ENGINE=1 \
-DWITH_MYISAM_STORAGE_ENGINE=1 \
-DMYSQL_UNIX_ADDR=%{DATA_DIR}/mysqld.sock \
-DMYSQL_TCP_PORT=%{MYSQL_PORT} \
-DENABLED_LOCAL_INFILE=1 \
-DEXTRA_CHARSETS=all \
-DDEFAULT_CHARSET=utf8mb4 \
-DDEFAULT_COLLATION=utf8mb4_general_ci \
-DWITH_EMBEDDED_SERVER=1 \
-DCPACK_MONOLITHIC_INSTALL=1

make %{?_smp_mflags} -j `cat /proc/cpuinfo | grep processor| wc -l`

%install
rm -rf %{buildroot}
make install DESTDIR=%{buildroot}
%{__install} -p -D %{SOURCE1} %{buildroot}%{BASE_DIR}/etc/common_schema-2.2-TD.sql

%pre
check_port() {
        netstat -nltup | grep -v "127.0.0.1" | grep $1 &>/dev/null
}
if [ -d %{DATA_DIR} ];then
        echo "Error: The data directory %{DATA_DIR} already exists, Aborting." ;
        exit 1
elif [ -d %{BASE_DIR} ];then
        echo "Error: The base directory %{BASE_DIR} already exists, Aborting." ;
        exit 1
elif check_port %{MYSQL_PORT}
then
        echo "Error: The port %{MYSQL_PORT} already in used, Aborting." ;
        exit 1
elif [ -e /etc/init.d/mysql%{MYSQL_PORT} ];
then
        echo "Error: The file /etc/init.d/mysql%{MYSQL_PORT} already exists, Aborting." ;
        exit 1
elif [ -e /lib/systemd/system/mysql%{MYSQL_PORT}.service ];
then
        echo "Error: The file /lib/systemd/system/mysql%{MYSQL_PORT}.service already exists, Aborting." ;
        exit 1
elif [ -e /etc/my.cnf ];
then
        echo "Error: The file /etc/my.cnf already exists, Aborting." ;
        exit 1
fi
TOTAL_SIZE=$( free -g | grep Mem | awk '{print $2}' )
ALLOC_SIZE=$(( $TOTAL_SIZE * 50 / 100 ))
BUFFER_SIZE=$( echo $ALLOC_SIZE | awk -F. '{print $1}')"G"
SERVER_ID=$( ip addr | grep inet | grep "scope global" | grep -v "secondary" |head -1|awk '{print $2}'|cut -d "." -f 4|cut -d "/" -f 1 )"%{MYSQL_PORT}"
mkdir -p %{BASE_DIR}/etc
cat > %{BASE_DIR}/etc/my.cnf <<EOF
[client]
port = %{MYSQL_PORT}
socket = %{DATA_DIR}/mysqld.sock
default-character-set=utf8mb4

[mysqld]
character-set-server=utf8mb4
port = %{MYSQL_PORT}
socket = %{DATA_DIR}/mysqld.sock
basedir = %{BASE_DIR}
datadir = %{DATA_DIR}
back_log = 500
skip-name-resolve
skip-external-locking
concurrent_insert = 2
delay-key-write = ALL
key_buffer_size = 256M
max_allowed_packet = 1024M
table_open_cache = 4096
sort_buffer_size = 2M
net_buffer_length = 8K
read_buffer_size = 2M
read_rnd_buffer_size = 512K
myisam_sort_buffer_size = 64M
thread_stack = 512K
log-bin = mysql-bin
log_bin_trust_function_creators = on
binlog_format = mixed
binlog_row_image=minimal
transaction-isolation = REPEATABLE-READ
sync_binlog=0
relay_log = relay-bin
server-id = $SERVER_ID
binlog_cache_size = 8M
max_binlog_size = 1024M
max_relay_log_size = 1024M
expire_logs_days = 15
thread_cache_size = 48
query_cache_type = on
query_cache_size = 0
query_cache_limit = 8M
max_connections = 4000
max_connect_errors = 2000
tmp_table_size = 256M
long_query_time = 3
slow_query_log=off
innodb_page_size = 16384
innodb_file_per_table
innodb_buffer_pool_size = $BUFFER_SIZE
innodb_flush_log_at_trx_commit = 2
innodb_flush_method = O_DIRECT
innodb_log_file_size = 1G
innodb_log_files_in_group = 3
innodb_fast_shutdown = 1
innodb_lock_wait_timeout=86400
sql_mode = NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTION,STRICT_TRANS_TABLES
innodb_io_capacity = 2000
innodb_io_capacity_max = 10000
innodb_open_files = 1000
federated
event_scheduler=1
secure-file-priv = ""
slave-parallel-type=LOGICAL_CLOCK
slave-parallel-workers=8
master_info_repository=TABLE
relay_log_info_repository=TABLE
relay_log_recovery=ON
explicit_defaults_for_timestamp=OFF
gtid-mode=ON
enforce-gtid-consistency=ON
slave_pending_jobs_size_max=512M
log_timestamps=SYSTEM
default-time-zone='+8:00'
log_slave_updates=0
open_files_limit=655350
show_compatibility_56=on
group_concat_max_len=10240000
optimizer_switch='mrr=on,mrr_cost_based=off'
innodb_undo_log_truncate=on
innodb_undo_tablespaces=3
innodb_max_undo_log_size=10737418240
[mysqldump]
quick
max_allowed_packet = 16M

[mysql]
no-auto-rehash

[myisamchk]
key_buffer_size = 20M
sort_buffer_size = 20M
read_buffer = 2M
write_buffer = 2M

[mysqlhotcopy]
interactive-timeout
EOF

mkdir -p %{DATA_DIR}
useradd -s /bin/nologin -M %{MYSQL_USER} >/dev/null 2>&1
chown -R %{MYSQL_USER}:%{MYSQL_GROUP} %{DATA_DIR}

%post
ln -sf %{BASE_DIR}/support-files/mysql.server /etc/init.d/mysql%{MYSQL_PORT}
ln -sf %{BASE_DIR}/bin/mysql /usr/bin/mysql%{MYSQL_PORT}

%{BASE_DIR}/bin/mysqld --defaults-file=%{BASE_DIR}/etc/my.cnf --initialize-insecure --user=mysql --basedir=%{BASE_DIR} --datadir=%{DATA_DIR} >/dev/null 2>&1

printf "[Unit]\nDescription=mysql%{MYSQL_PORT}\n[Service]\nType=forking\nUser=mysql\nGroup=mysql\nExecStart=/etc/init.d/mysql%{MYSQL_PORT} start\nExecReload=/etc/init.d/mysql%{MYSQL_PORT} restart\nExecStop=/etc/init.d/mysql%{MYSQL_PORT} stop\nPrivateTmp=true\nTimeoutSec=0\nLimitNOFILE=infinity\nLimitMEMLOCK=infinity\n[Install]\nWantedBy=multi-user.target\n">/lib/systemd/system/mysql%{MYSQL_PORT}.service

systemctl enable mysql%{MYSQL_PORT}.service >/dev/null 2>&1

systemctl daemon-reload

chown -R %{MYSQL_USER}:%{MYSQL_GROUP} %{DATA_DIR}

chmod -R og-rw %{DATA_DIR}

if [ -e /etc/my.cnf ];then
        mv /etc/my.cnf /etc/my.cnf_bak_`date +%Y%m%d%H%M%S`
fi

systemctl start mysql%{MYSQL_PORT}.service

sleep 10

#if [ -x /usr/bin/openssl ] ; then
#        passwd=`/usr/bin/openssl rand -base64 8 | cut -c 1-8`
#else
#        passwd=`/usr/bin/cat /dev/urandom | head -n 8 | md5sum | head -c 8`
#fi

sql=`/usr/bin/echo "
drop database if exists test;
DELETE FROM mysql.user where user='';
DELETE FROM mysql.user where user='root' and host<>'localhost';
GRANT ALL PRIVILEGES ON *.* TO root@localhost identified by PASSWORD '*EEEDEED0EEE5735BC3FC11AADF97B497233B6FEE' WITH GRANT OPTION;
flush privileges;
reset master;"`

common_schema_sql=`/usr/bin/echo "source %{BASE_DIR}/etc/common_schema.sql;"`

/usr/local/mysql%{MYSQL_PORT}/bin/mysql -uroot -q -s -S %{DATA_DIR}/mysqld.sock -e "${common_schema_sql}"

/usr/local/mysql%{MYSQL_PORT}/bin/mysql -uroot -q -s -S %{DATA_DIR}/mysqld.sock -e "${sql}"

echo "Info: MySQL %{version} install successed, run 'mysql3306 -uroot -p****** -P%{MYSQL_PORT} -hlocalhost' to connect!"

%preun
systemctl stop mysql%{MYSQL_PORT}.service
systemctl disable mysql%{MYSQL_PORT}.service
sleep 10s

%postun
rm -f /lib/systemd/system/mysql%{MYSQL_PORT}.service
rm -rf /usr/local/mysql%{MYSQL_PORT}
rm -rf /mysql/mysql%{MYSQL_PORT}
rm -f /etc/init.d/mysql%{MYSQL_PORT}
rm -f /usr/bin/mysql%{MYSQL_PORT}

%clean
rm -rf %{buildroot}

%files
%defattr(-, %{MYSQL_USER}, %{MYSQL_GROUP})
%attr(755, %{MYSQL_USER}, %{MYSQL_GROUP}) %{BASE_DIR}/*

%changelog

    不需要关闭服务直接卸载：

    [root@rhce ~]# rpm -e --nodeps mysql-5.7.17-1.el6.x86_64

    注意这里配置了，在卸载服务时会把mysql目录删除，所以要先备份数据。

    # rpm相关命令

    • rpm -qp --scripts mysql-5.5.32-1.x86_64.rpm - 查看包内脚本信息
    • rpm -qpl mysql-5.5.32-1.x86_64.rpm - 查看rpm包内包含哪些文件及目录

    # 编译命令说明

    rpmbuild

    • -ba 既生成src.rpm又生成二进制rpm
    • -bs 只生成src的rpm
    • -bb 只生二进制的rpm
    • -bp 执行到pre
    • -bc 执行到 build段
    • -bi 执行install段
    • -bl 检测有文件没包含
